##Установка git
##Регистрация/авторизация
##Создание репозитория ##Команда для создания репозитория Git — git init. Эта команда инициализирует пустой репозиторий.
git clone	##Копирование репозитория
git add File1.txt ##Указание на существование файла
git status ##Просмотр папки git
git commit -m "commit файла" ##Коммит файла
git commit -am "commit всех файлов" ##Коммит всех файлов
git reset HEAD~1 ##Отмена коммита на шаг назад
git push ##Загрузка изменений в сеть
cat File1.txt ##Просмотр файла
git fetch ##это команда, которая используется для извлечения всех изменений из удалённого репозитория в локальный репозиторий.Она не вносит изменений в текущий рабочий каталог и сохраняет все изменения в отдельной ветке, называемой веткой удалённого отслеживания.Для слияния этих изменений в текущий рабочий каталог используется команда git merge или git rebase.
##origin main - версия ветки на удаленном репозитории
##origin/main - локальная, промежуточная ветка
git pull ##используется для извлечения изменений из удалённого репозитория и автоматического объединения их с текущей веткой. По сути, git pull — это комбинация команд git fetch и git merge
git log ##Просмотр всех commit
git log --author gz5658 ##Просмотр commit определенного автора
git show номер коммита ##Просмотр определенного коммита
git blame File1.txt ##История commit по авторам
git blame File1.txt | grep "Создание файла" ##Показывает кто ввел эту строку в файл
git blame File1.txt | grep GZ5658 ##Показывает какие изменения ввел этот автор
echo "Add Line" >> File1.txt ##Добавление строки в файл
git diff ##Показывает какие изменения были введены
ls ##Просмотр файлов в репозитории
echo "new line" >> new.file ##Создание нового файла с текстом
git checkout File1.txt ##используется для переключения между ветками, проверки коммитов и отката изменений
git stash ##позволяет записывать изменения в коде в локальное хранилище и возвращаться к ним позже
git stash pop ## применяет изменения из последнего сохранения в стеке stash к текущему рабочему каталогу и удаляет это сохранение из стека
git stash clear ##удаляет все сохранённые в git stash состояния
